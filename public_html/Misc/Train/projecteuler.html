    return 1

# squre_sum = n(n+1)(2n+1)/6
# sum = (1+n)*n/2
def diff_num(n):
    sum = (1+n)*n/2
    squre_sum = n*(n+1)*(2*n+1)/6
    return sum**2 - squre_sum

print diff_num(100)
